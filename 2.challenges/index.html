Solucion: 
        
          function reverseString(str) {
            return str === ''
              ? ''
              : reverseString(str.substr(1)) + str.charAt(0);

            return list.sort(function(a, b) {

d. Crear una función que tenga como argumento el siguiente string y que el retorno de esa función sea el mismo string en mayúscula (sin usar el método toUpperCase()).

        
          const stringTest = 'fintech';
Solucion: 
        
          function upperCaseString(str) {
            return str === ''
              ? ''
              : str.charCodeAt(0) >= 97 && str.charCodeAt(0) <= 122
              ? String.fromCharCode(str.charCodeAt(0) - 32) +
                upperCaseString(str.substr(1))
              : str.charAt(0) + upperCaseString(str.substr(1));
          }

e. Muestre las tablas de multiplicar del 1 al 10 utilizando sólo un for.

Solucion: 
        
          function multiplicationTables(limit) {
            if (limit === 0) {
              return '';
            } else {
              var t = '';
              for (var index = 1; index <= 10; index++) {
                t = `${t}
                     ${limit}*${index} = ${index * limit}`;
              }
              return `${multiplicationTables(limit - 1)}
                      ${t}`;
            }
          }

f. Función recursiva para obtener el factorial del siguiente número.

        
          const numberTest = 43;
Solucion: 
        
          function factorial(num) {
            num = Math.abs(num);
            if (num === 1) {
              return 1;
            } else {
              return num * factorial(num - 1);
            }
          }
